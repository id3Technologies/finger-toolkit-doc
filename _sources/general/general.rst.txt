Adding the library to your project
==================================

.. tab-set::

    .. tab-item:: C
        :sync: c

        .. code-block:: c

            #include "id3Finger.h"

    .. tab-item:: Dart
        :sync: dart

        .. code-block:: dart

            import 'package:id3_finger/id3_finger.dart';

    .. tab-item:: C#
        :sync: csharp

        .. code-block:: csharp

            using id3.Finger;

    .. tab-item:: Java
